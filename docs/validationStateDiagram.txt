  └── VALIDATE_INPUT
        ├── isEmptyInput(address)?
        │     └── YES → EMPTY_INPUT
        └── NO → VALIDATE_ADDRESS
               └── isValidAddress(address)?
                     ├── NO  → INVALID_ADDRESS_INPUT
                     └── YES → TEST_DUPLICATE_INPUT
                            └── isDuplicateInput(containerType)?
                                  ├── YES → DUPLICATE_INPUT
                                  └── NO  → VALIDATE_EXISTS_ON_CHAIN
                                         └── publicClient available?
                                               ├── NO  → warn + skip
                                               └── YES → resolveTokenContract()
                                                     ├── not found → CONTRACT_NOT_FOUND_ON_BLOCKCHAIN
                                                     └── found
                                                           ├── setValidatedAsset()
                                                           ├── setTokenContract (context)
                                                           └── VALIDATE_CONTRACT_EXISTS_LOCALLY
                                                                 └── hasBrokenLogoURL?
                                                                       ├── YES → CONTRACT_NOT_FOUND_LOCALLY
                                                                       └── NO  → IS_LOADING
                                                                              └── wait for balanceData
                                                                                    └── when ready:
                                                                                          ├── enrich validatedAsset
                                                                                          ├── setValidatedAsset()
                                                                                          ├── setTokenContract (context)
                                                                                          └── VALID_INPUT
                                                                                                └── promote → CLOSE_SELECT_INPUT
                                                                                                      ├── set token in context
                                                                                                      └── EMPTY_INPUT
